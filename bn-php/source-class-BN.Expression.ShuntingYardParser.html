 20:  * Parser for parsing mathematical expressions from infix to postfix notation (RPN).
 21:  * It's implementation of Dijkstra Shunting Yard algorithm. Result of parsing is
 22:  * queue with operands and operators (brackets are not needed in postfix notation)
 23:  */

 74:      * Sets operators used in expression. Every operator (value in array) have to

 90:     /**
 91:      * Sets variables which can be used in expression. Key from associative array
 92:      * is used as variable name and value is used for replacing name during parsing.
 93:      * Variables are appended to existing variables and if two variables have same
 94:      * name then the last time variable operator is used.
 95:      * @example array('A' => new BN\Number('555')); and expression can be 'A + 5'
 96:      * when result will be 560
 97:      * @param  array                     $variables
 98:      * @throws \InvalidArgumentException if variable name (array) is not string or if it's numeric
 99:      */
100:     public function setVariables(array $variables)
101:     {
102:         foreach ($variables as $variableName => $value) {
103:             if ($this->isVariableNameValid($variableName)) {
104:                 $this->variables[$variableName] = $value;
105:             } else {
106:                 throw new \InvalidArgumentException('Variable name (array key) must be non-numeric string');
107:             }
108:         }
109:     }

124:     /**
125:      * Parse expression with Shunting Yard algorithm
126:      * @param  string                $expression
127:      * @return \BN\Collections\Queue
128:      * Returns queue with operators and numbers. Queue is representation of input
129:      * mathematical expression in postfix notation (RPN)
130:      * @throws \BN\Expression\Exception\ParserException if $expression is not string or invalid tokenizer
131:      * or invalid value in expression
132:      * @throws \BN\Expression\Exception\MismatchedBracketException
133:      */
134:     public function parse($expression)
135:     {
136:         $this->queue->clear();
137:         $this->stack->clear();
138:         $this->tokenizer->tokenizeExpression($expression);
139:         foreach ($this->tokenizer->getIterator() as $token) {
140:             $this->parseToken($token);
141:         }
142:         while (!$this->stack->isEmpty()) {
143:             $this->pushOperatorFromStack();
144:         }
145:         return clone $this->queue;
146:     }
147: 
148:     /**
149:      * Parsing of token from expression
150:      * @param  string          $token
151:      * @throws ParserException if tokenizer produces invalid token
152:      */
153:     private function parseToken($token)
154:     {
155:         if (!is_string($token)) {
156:             throw new ParserException('Tokenizer must break expression into strings');
157:         } elseif (isset($this->operators[$token])) {
158:             $this->parseOperator($this->operators[$token]);
159:         } elseif ($this->brackets->isLeftBracket($token)) {
160:             $this->parseLeftBracket();
161:         } elseif ($this->brackets->isRightBracket($token)) {
162:             $this->parseRightBracket();
163:         } else {
164:             $this->parseNumber($token);
165:         }
166:     }
167: 
168:     /**
169:      * While there is an operator token, at the top of the stack, and either $operator
170:      * is left-associative and its precedence is less than or equal to that of top
171:      * of the stack, or $operator has precedence less than that of top of the stack,
172:      * pop top of the stack off the stack, onto the output queue.
173:      * Then $operator is pushed onto the stack
174:      * @param \BN\Expression\Operator\IOperator $operator
175:      */
176:     private function parseOperator(IOperator $operator)
177:     {
178:         if (!$this->stack->isEmpty()) {
179:             while ($this->isOperatorAtTopOfStack()) {
180:                 $pushOperator = $this->canBeOperatorPushedFromStack($operator);
181:                 if ($pushOperator) {
182:                     $this->queue->push($this->stack->pop());
183:                 }
184:                 if ($this->stack->isEmpty() || $pushOperator == false) {
185:                     break;
186:                 }
187:             }
188:         }
189:         $this->stack->push($operator);
190:     }
191: 
192:     private function isOperatorAtTopOfStack()
193:     {
194:         return $this->stack->peek() instanceof IOperator;
195:     }
196: 
197:     private function canBeOperatorPushedFromStack($operator)
198:     {
199:         $isLeftAssociative = $operator->getAssociativity() == Associativity::LEFT;
200:         $hasLowerOrEqualPrecedence = $operator->getPrecedence() <= $this->stack->peek()->getPrecedence();
201:         $hasLowerPrecedence = $operator->getPrecedence() < $this->stack->peek()->getPrecedence();
202:         return ($isLeftAssociative && $hasLowerOrEqualPrecedence) || $hasLowerPrecedence;
203:     }
204: 
205:     /**
206:      * Push left bracket onto the stack
207:      */
208:     private function parseLeftBracket()
209:     {
210:         $this->stack->push($this->brackets->getLeftBracket());
211:     }
212: 
213:     /**
214:      * While the token at the top of the stack is not a left bracket, pop operators off
215:      * the stack onto the output queue. Then it pop the left bracket from the stack,
216:      * but not onto the output queue.
217:      * @throws MismatchedBracketException if there is no left bracket in stack
218:      */
219:     private function parseRightBracket()
220:     {
221:         while ($this->isNotLeftBracketAtTopOfStack()) {
222:             if ($this->stack->isEmpty()) {
223:                 throw new MismatchedBracketException();
224:             }
225:             $this->queue->push($this->stack->pop());
226:         }
227:         $this->stack->pop();
228:     }
229: 
230:     private function isNotLeftBracketAtTopOfStack()
231:     {
232:         return !$this->brackets->isLeftBracket($this->stack->peek());
233:     }
234: 
235:     /**
236:      * If the token is a numeric or defined variable, then push it to the output queue.
237:      * @param  string                    $token
238:      * @throws \InvalidArgumentException if $token is not numeric and variable
239:      * with that name is not defined
240:      */
241:     private function parseNumber($token)
242:     {
243:         if ($this->isNotVariable($token)) {
244:             $this->defineVariable($token);
245:         }
246:         $this->queue->push($this->variables[$token]);
247:     }
248: 
249:     private function isNotVariable($token)
250:     {
251:         return !isset($this->variables[$token]);
252:     }
253: 
254:     private function defineVariable($token)
255:     {
256:         if (is_numeric($token)) {
257:             $this->variables[$token] = new \BN\Number($token);
258:         } else {
259:             throw new ParserException('Invalid value in expression');
260:         }
261:     }
262: 
263:     /**
264:      * Used when no more tokens are available and the stack is not empty.
265:      * If operator on the top of the stack is operator then push it onto the
266:      * output queue. If a bracket is on the top of the stack, then there are
267:      * mismatches brackets
268:      * @throws MismatchedBracketException
269:      */
270:     private function pushOperatorFromStack()
271:     {
272:         if ($this->isBracketAtTopOfStack()) {
273:             throw new MismatchedBracketException();
274:         } elseif ($this->isOperatorAtTopOfStack()) {
275:             $this->queue->push($this->stack->pop());
276:         }
277:     }
278: 
279:     private function isBracketAtTopOfStack()
280:     {
281:         return $this->brackets->isBracket($this->stack->peek());
282:     }

 12:     /**
 13:      * Method to save event to the log file. Location of logfile is determined 
 14:      * from constants which are defined in the file logger.php.<br />
 15:      * @param mixed $event - any instance of class which has implemented __toString() method
 16:      * @param string $logname
 17:      * @return boolean 
 18:      * Returns true if message was successfully saved. False if: log's folder
 19:      * doesn't exist or missing file permissions to log file.
 20:      */
 21:     public static function logEvent($event, $logname = TREEWEC_LOGGER_DEFAULT_LOG_NAME) {
 22:         if (self::isEvent($event) && self::isLogWritable($logname)) {
 23:             $file = fopen(TREEWEC_LOGGER_LOG_ROOT . $logname . TREEWEC_LOGGER_EXT, 'a');
 24:             if ($file) {
 25:                 fwrite($file, $event);
 26:                 fclose($file);
 27:                 return true;
 28:             }
 29:         }
 30:         return false;
 31:     }

 33:     /**
 34:      * Converts csv file into 2D array.
 35:      * @param string $logname
 36:      * @return array  
 37:      * Returns 2-dimensional array, if file from argument is readable (exists)
 38:      * Example: 
 39:      *  $array[0] = array(21, 11, 1989, 'Login', 'Zdenek Drahos');
 40:      *  $array[0][1] = 11;        
 41:      */
 42:     public static function getLogContent($logname = TREEWEC_LOGGER_DEFAULT_LOG_NAME) {
 43:         if (is_string($logname) && is_readable(TREEWEC_LOGGER_LOG_ROOT . $logname . TREEWEC_LOGGER_EXT)) {
 44:             $array = file(TREEWEC_LOGGER_LOG_ROOT . $logname . TREEWEC_LOGGER_EXT);
 45:             for ($index = 0; $index < count($array); $index++) {
 46:                 $array[$index] = explode(TREEWEC_LOGGER_SEP, $array[$index]);
 47:             }
 48:             return $array;
 49:         }
 50:         return false;
 51:     }
 52: 
 53:     /**
 54:      * Returns only last X events from logfile.
 55:      * Events are sorted from the latest to the oldest X events.
 56:      * @param int $count 
 57:      * @param string $logname
 58:      * @return array
 59:      * Returns array if $count is integer and $logname is existing file in TREEWEC_LOGGER_LOG_ROOT.
 60:      * If $count is smaller than one or bigger than number of records in file then 
 61:      * returns full content of logfile (in reverse order)
 62:      */
 63:     public static function getLastEvents($count, $logname = TREEWEC_LOGGER_DEFAULT_LOG_NAME) {
 64:         $log = self::getLogContent($logname);
 65:         if (is_array($log) && is_int($count)) {
 66:             $log_count = count($log);
 67:             if (self::isCountCorrect($count, $log_count)) {
 68:                 $last_events = array();
 69:                 for ($i = ($log_count - $count); $i < $log_count; $i++) {
 70:                     $last_events[] = $log[$i];
 71:                 }
 72:                 $log = $last_events;
 73:             }
 74:             return array_reverse($log);
 75:         }
 76:         return false;
 77:     }
 78: 
 79:     private static function isCountCorrect($count, $log_count) {
 80:         return $count > 0 && $count < $log_count;
 81:     }

 97:      * Finds whether a variable can be writed into the log file.
 98:      * @param object $event
 99:      * @return boolean
100:      * True if $event is object and implements __toString()), otherwise false
101:      */
102:     private static function isEvent($event) {
103:         return is_object($event) ? method_exists($event, '__toString') : false;
104:     }

106:     /**
107:      * Finds whether an event can be saved into the log file and if $logname
108:      * can be used as name of the file
109:      * @param string $logname 
110:      * @return boolean 
111:      * Returns false, if log folder doesn't exist or user don't have file permission to write
112:      * or $logname is not string
113:      */
114:     private static function isLogWritable($logname = TREEWEC_LOGGER_DEFAULT_LOG_NAME) {
115:         if (is_string($logname) && file_exists(TREEWEC_LOGGER_LOG_ROOT)) {
116:             $file_exists = file_exists(TREEWEC_LOGGER_LOG_ROOT . $logname . TREEWEC_LOGGER_EXT);
117:             return ($file_exists && is_writable(TREEWEC_LOGGER_LOG_ROOT . $logname . TREEWEC_LOGGER_EXT)) ||
118:                     (!$file_exists && is_writable(TREEWEC_LOGGER_LOG_ROOT));
119:         }
120:         return false;
121:     }
